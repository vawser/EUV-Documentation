#-----------------------------------
# Iterator Effects
#-----------------------------------
The three types iterator effects can use these parameters:

every_<list> = {
    # Filters the list based on the triggers
    limit = {
        <triggers>
    }
    
    <effects>
}

ordered_<list> = {
    # Filters the list based on the triggers
    limit = {
        <triggers>
    }
    
    order_by = <value>
    
    position = <value>  # Optional
    min = <value>       # Optional
    max = <value>       # Optional
    check_range_bounds = <bool> # If you don't want an error logged if the list is smaller than the min/max
    
    <effects>
}


random_<list> = {
    # Filters the list based on the triggers
    limit = {
        <triggers>
    }
    
    weight = {
        <min mtth>
        <max mtth>
    }
    # Optional
    
    <effects>
}

The -> indicates the inner scope the iterator targets.

Only the every_ scope is noted below, assume the ordered_ and random_scopes exist too unless noted otherwise.
