#-----------------------------------
# Comparison Types
#-----------------------------------
Less Than: <
Less Than or Equal: <=
Equals: =
Not Equals: !=
Greater Than: >
Greater Than or Equal: >=

For value comparisons, these comparitors can be used. 

#-----------------------------------
# Triggers
#-----------------------------------
For some triggers, the scope parameters can be condensed, for example:

num_of_trait_by_category = {
    type = <trait_category>
    value <comparator> <integer>
}

Can be used as such:

num_of_trait_by_category(<trait_category>) = <value>

# Scope: None
#--------------------------------------------------------------------------------
# ----- Flow -----
Flow triggers are used to control how other triggers are executed.

all_false = { 
    <triggers> 
}
# true if all children are false (equivalent to NOR) 

and = { 
    <triggers> 
}
# all inside trigger must be true 

any_false = { 
    <triggers> 
}
# true if any child is false (equivalent to NAND) 

calc_true_if = {
    amount = <value>
    
    <triggers>
}
# Returns true if the specified number of sub-triggers return true 

custom_description = {
    text = <loc key>
    subject = <scope> # optional
    object = <scope> # optional
    value = <script value> # optional
    
    <triggers>
}
# Wraps triggers that get a custom description instead of the auto-generated one 

hidden_trigger = {
    <triggers>
}
# Enclosed triggers are not shown in tooltips
