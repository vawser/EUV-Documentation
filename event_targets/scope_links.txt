#--------------------------------------------------------------------------------
# Scope Links
#--------------------------------------------------------------------------------
These are basic scope switches. -> points to the new scope.

#----------------------
# Scope: None
#----------------------
c -> country
# Scope to the specified country TAG 	 

#----------------------
# Scope: Building
#----------------------
building_type -> building_type
# Scope to the building type this building belongs to.

location -> location
# Scope to the location for this building

owner -> country
# Scope to the owner country for this building

#----------------------
# Scope: Character
#----------------------
birth_location -> location
# Scope to the birth location of this character.

cabinet_action -> cabinet_action
# Scope to the cabinet action this character is performing.

dynasty -> dynasty
# Scope to dynasty this character belongs to.

employer -> country
# Scope to the employing country of this character.

ethnicity -> ethnicity
# Scope to ethnicity group this character belongs to.

exploration -> exploration
# Scope to exploration group this character belongs to.

father -> character
# Scope to the father of this character.

first_spouse -> character
# Scope to the first spouse of this character.

location -> location
# Scope to the location of this character

mother -> character
# Scope to the mother of this character.

owner -> country
# Scope to country that this character is owned by.

rebel -> rebels
# Scope to the rebel group this character belongs to.

religious_school -> religious_school
# Scope to the religious school this character belongs to.

rule_end_date -> date
# Scope to the date that this character's rule ended at

unit -> unit
# Scope to the unit this character is in.

#----------------------
# Scope: Country

# Scope to the civil war this country is under going.

# Scope to the opponent country in the civil war this country is under going.

consort -> character
# Scope to the consort of the ruler character that this country has.

# Scope to the color for this country.

# Scope to the country rank on date for this country.

country_stance 	-> military_stance
# Scope to the current military stance for this country.

court_dialect -> dialect
# Scope to the current court dialect for this country.

# Scope to the current court language for this country.

current_mission_task -> mission_task
# Scope to the current mission task for this country.

dominant_culture -> culture
# Scope to the dominant culture for this country.

dominant_dialect -> dialect
# Scope to the dominant dialect for this country.

dominant_language -> language
# Scope to the dominant language for this country.

dominant_religion -> religion
# Scope to the dominant religion for this country.

dominant_upper_class_culture -> culture
# Scope to the dominant upper-class culture for this country.

estate -> estate
# Scope to a particular estate for this country.
# Usage: estate:<estate_type_link> or estate(<estate_type_link>) 	

heir -> character
# Scope to the current heir for the ruler of this country.

last_valid_ruler -> character
# Scope to the last valid ruler of this country.

law_policy -> policy
# Scope to the policy chosen for a particular law.
# Usage: law_policy(<law>) 	

liturgical_dialect -> dialect
# Scope to the religious dialect of this country.

# Scope to the religious language of this country.

overlord -> country

# Scope to the previous ruler of this country.

# Scope to the regent of this country.

# Scope to the religious school this country follows.

ruler -> character
# Scope to the ruler of this country.

ruler_or_heir_if_regent -> character
# Scope to the ruler of this country, or the heir if in a regency.

ruler_or_regent -> character
# Scope to the ruler of this country, or the regent if in a regency.

succession_law -> heir_selection
# Scope to the succession law of this country.

top_overlord -> country
# Scope to the top overlord in the subject hierarchy of this country.

top_overlord_or_this -> country
# Scope to the top overlord in the subject hierarchy of this country, or this country if it is that overlord.

union -> international_organization
# Scope to the Union IO for this country.

war_with_country -> war
# Scope to the country at war with this country.
# Usage: war_with_country(<country>)

# Scope to on-going sieges at this location
