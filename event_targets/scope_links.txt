#--------------------------------------------------------------------------------
# Scope Links
#--------------------------------------------------------------------------------
These are basic scope switches. -> points to the new scope.

#----------------------
# Scope: None
#----------------------
c -> country
# Scope to the specified country TAG 	 

#----------------------
# Scope: Building
#----------------------
building_type -> building_type
# Scope to the building type object for this building

location -> location
# Scope to the location for this building

owner -> country
# Scope to the owner country for this building
